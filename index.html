    <div class="max-w-6xl mx-auto">
        <!-- Header -->
        <header class="flex flex-col md:flex-row justify-between items-start md:items-center mb-10 gap-4">
            <div>
                <h1 class="text-3xl md:text-4xl font-bold tracking-tight mb-1">Halo, Moe! ✨</h1>
                <p class="text-stone-500 dark:text-stone-400 font-medium" id="current-date">Memuat tanggal...</p>
            </div>
            
            <div class="flex items-center gap-4">
                <div class="glass-card px-4 py-2 font-mono text-sm tracking-widest hidden md:block" id="live-clock">
                    00:00
                </div>
                <button onclick="toggleTheme()" class="glass-card p-3 rounded-full hover:bg-stone-100 dark:hover:bg-stone-800 transition-colors" aria-label="Toggle Dark Mode">
                    <i class="ph ph-moon text-xl block dark:hidden"></i>

        <!-- Main Content Grid -->

    <!-- Application Logic -->
    <script>
        // --- State Management (In-Memory for single session) ---
        let tasks = [];
        let moments = [];

        // --- Theme Management ---
        function toggleTheme() {
            const html = document.documentElement;
            if (html.classList.contains('dark')) {
                html.classList.remove('dark');
            } else {
                html.classList.add('dark');
            }
        }

        // --- Time & Date Functions ---
        function updateDateTime() {
            const now = new Date();
            
            // Date formatting
            const optionsDate = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };
            document.getElementById('current-date').innerText = now.toLocaleDateString('id-ID', optionsDate);
            
            // Time formatting
            const hours = String(now.getHours()).padStart(2, '0');
            const minutes = String(now.getMinutes()).padStart(2, '0');
            document.getElementById('live-clock').innerText = `${hours}:${minutes}`;
        }
        
        setInterval(updateDateTime, 1000);
        updateDateTime(); // Initial call

        // --- Task Management ---
        const taskForm = document.getElementById('task-form');
        const taskContainer = document.getElementById('task-container');

        taskForm.addEventListener('submit', (e) => {
            e.preventDefault();
            const name = document.getElementById('task-name').value;
            const time = document.getElementById('task-time').value;
            const category = document.getElementById('task-category').value;

            const newTask = {
                id: Date.now().toString(),
                name,
                time,
                category,
                completed: false
            };

            tasks.push(newTask);
            // Sort tasks by time
            tasks.sort((a, b) => a.time.localeCompare(b.time));
            
            renderTasks();
            taskForm.reset();
        });

        function toggleTaskComplete(id) {
            const task = tasks.find(t => t.id === id);
            if (task) {
                task.completed = !task.completed;
                renderTasks();
            }
        }

        function deleteTask(id) {
            tasks = tasks.filter(t => t.id !== id);
            renderTasks();
        }

        function getCategoryColor(category) {
            switch(category) {
                case 'Rutinitas': return 'bg-blue-100 text-blue-700 dark:bg-blue-900/30 dark:text-blue-300 border-blue-200 dark:border-blue-800';
                case 'Hobi': return 'bg-orange-100 text-orange-700 dark:bg-orange-900/30 dark:text-orange-300 border-orange-200 dark:border-orange-800';
                case 'Me Time': return 'bg-purple-100 text-purple-700 dark:bg-purple-900/30 dark:text-purple-300 border-purple-200 dark:border-purple-800';
                case 'Kerja': return 'bg-stone-200 text-stone-700 dark:bg-stone-800 dark:text-stone-300 border-stone-300 dark:border-stone-700';
                default: return 'bg-stone-100 text-stone-600 dark:bg-stone-800/50 dark:text-stone-400 border-stone-200 dark:border-stone-700';
            }
        }

        function renderTasks() {
            if (tasks.length === 0) {
                taskContainer.innerHTML = `
                    <div class="text-center py-8 px-4 border border-dashed border-stone-300 dark:border-stone-700 rounded-3xl">
                        <i class="ph ph-coffee text-4xl text-stone-300 dark:text-stone-600 mb-2"></i>
                        <p class="text-stone-500 dark:text-stone-400 text-sm">Belum ada jadwal. Yuk rencanakan hari ini!</p>
                    </div>
                `;
                return;
            }

            taskContainer.innerHTML = tasks.map(task => `
                <div class="glass-card p-4 flex items-center gap-4 group hover:shadow-md transition-shadow">
                    <button onclick="toggleTaskComplete('${task.id}')" class="flex-shrink-0 w-6 h-6 rounded-full border-2 ${task.completed ? 'bg-stone-800 border-stone-800 dark:bg-stone-200 dark:border-stone-200 text-white dark:text-stone-900' : 'border-stone-400 dark:border-stone-500'} flex items-center justify-center transition-colors">
                        ${task.completed ? '<i class="ph ph-check text-xs font-bold"></i>' : ''}
                    </button>
                    
                    <div class="flex-grow min-w-0">
                        <p class="font-semibold text-lg truncate ${task.completed ? 'line-through text-stone-400 dark:text-stone-600' : ''}">${task.name}</p>
                        <div class="flex items-center gap-3 mt-1 text-xs">
                            <span class="flex items-center gap-1 font-mono text-stone-500 dark:text-stone-400">
                                <i class="ph ph-clock"></i> ${task.time}
                            </span>
                            <span class="px-2 py-0.5 rounded-full border ${getCategoryColor(task.category)}">
                                ${task.category}
                            </span>
                        </div>
                    </div>
                    
                    <button onclick="deleteTask('${task.id}')" class="opacity-0 group-hover:opacity-100 text-red-400 hover:text-red-500 dark:text-red-500 dark:hover:text-red-400 p-2 transition-opacity focus:opacity-100" aria-label="Hapus">
                        <i class="ph ph-trash text-lg"></i>
                    </button>
                </div>
            `).join('');
        }

        // --- Moment Management ---
        const momentForm = document.getElementById('moment-form');
        const momentContainer = document.getElementById('moment-container');

        momentForm.addEventListener('submit', (e) => {
            e.preventDefault();
            const text = document.getElementById('moment-text').value;
            const now = new Date();
            const timeString = `${String(now.getHours()).padStart(2, '0')}:${String(now.getMinutes()).padStart(2, '0')}`;

            const newMoment = {
                id: Date.now().toString(),
                text,
                time: timeString
            };

            // Add to beginning of array
            moments.unshift(newMoment);
            renderMoments();
            momentForm.reset();
        });

        function deleteMoment(id) {
            moments = moments.filter(m => m.id !== id);
            renderMoments();
        }

        function renderMoments() {
            if (moments.length === 0) {
                momentContainer.innerHTML = `
                    <div class="text-center py-8 px-4 border border-dashed border-stone-300 dark:border-stone-700 rounded-3xl">
                        <i class="ph ph-notebook text-4xl text-stone-300 dark:text-stone-600 mb-2"></i>
                        <p class="text-stone-500 dark:text-stone-400 text-sm">Belum ada momen yang tercatat.</p>
                    </div>
                `;
                return;
            }

            momentContainer.innerHTML = moments.map(moment => `
                <div class="p-5 border border-stone-200 dark:border-stone-800 bg-white/40 dark:bg-stone-900/40 rounded-2xl relative group">
                    <button onclick="deleteMoment('${moment.id}')" class="absolute top-3 right-3 opacity-0 group-hover:opacity-100 text-stone-400 hover:text-red-400 transition-opacity" aria-label="Hapus">
                        <i class="ph ph-x"></i>
                    </button>
                    <p class="text-stone-700 dark:text-stone-300 leading-relaxed text-sm pr-6 italic">"${moment.text}"</p>
                    <p class="text-xs text-stone-400 dark:text-stone-500 font-mono mt-3 text-right">- ${moment.time}</p>
                </div>
            `).join('');
        }

        // Initial Render
        renderTasks();
        renderMoments();
    </script>
